346/// Thin wrapper around [`zbus::Connection`] that calls to `T`'s implementation
347/// of [`RootInterface`], [`PlayerInterface`], [`TrackListInterface`], and
348/// [`PlaylistsInterface`] to implement `org.mpris.MediaPlayer2` and its
349/// sub-interfaces.
350///
351/// When implementing using [`Server`], it is important to note that properties
352/// changed signals are *not* emitted automatically; they must be emitted
353/// manually using [`Server::properties_changed`],
354/// [`Server::track_list_properties_changed`], or
355/// [`Server::playlists_properties_changed`], when they changed internally.
356pub struct Server<T> {
357    connection: Connection,
358    imp: Arc<T>,
359    bus_name: OwnedWellKnownName,
360}
361
362impl<T> fmt::Debug for Server<T> {
363    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
364        f.debug_struct("Server").finish()
365    }
366}

368macro_rules! insert_property {
369    ($item:ident, $property_type:ident => $($map:ident, $property:ident);*) => {
370        match $item {
371            $(
372                $property_type::$property(val) => {
373                    $map.insert(stringify!($property), Value::new(val));
374                }
375            )*
376        }
377    };
378}

380impl<T> Server<T>
381where
382    T: PlayerInterface + 'static,
383{
384    /// Creates a new [`Server`] with the given bus name suffix and
385    /// implementation, `imp`, which must implement [`RootInterface`] and
386    /// [`PlayerInterface`].
387    ///
388    /// The resulting bus name will be
389    /// `org.mpris.MediaPlayer2.<bus_name_suffix>`, where
390    /// `<bus_name_suffix>`must be a unique identifier, such as one based on a
391    /// UNIX process id. For example, this could be:
392    ///
393    /// * `org.mpris.MediaPlayer2.vlc.instance7389`
394    ///
395    /// **Note:** According to the [D-Bus specification], the unique
396    /// identifier "must only contain  the ASCII characters
397    /// `[A-Z][a-z][0-9]_-`" and "must not begin with a digit".
398    ///
399    /// [D-Bus specification]: https://dbus.freedesktop.org/doc/dbus-specification.html#message-protocol-names-bus
400    pub async fn new(bus_name_suffix: &str, imp: T) -> Result<Self> {
401        Self::new_inner(bus_name_suffix, imp, |builder, _| Ok(builder)).await
402    }
403
404    /// Returns a reference to the underlying implementation.
405    #[inline]
406    pub fn imp(&self) -> &T {
407        &self.imp
408    }
409
410    /// Returns a reference to the inner [`Connection`].
411    ///
412    /// If you needed to call this, consider filing an issue.
413    #[cfg(feature = "unstable")]
414    #[cfg_attr(docsrs, doc(cfg(feature = "unstable")))]
415    #[inline]
416    pub fn connection(&self) -> &Connection {
417        &self.connection
418    }
419
420    /// Emits the given signal.
421    pub async fn emit(&self, signal: Signal) -> Result<()> {
422        match signal {
423            Signal::Seeked { position } => {
424                self.emit_inner::<RawPlayerInterface<T>>("Seeked", &(position,))
425                    .await?;
426            }
427        }
428
429        Ok(())
430    }
431
432    /// Emits the `PropertiesChanged` signal for the given properties.
433    ///
434    /// This categorizes the property in the `changed` or `invalidated`
435    /// properties as defined by the spec.
436    ///
437    /// [`Server::track_list_properties_changed`] or
438    /// [`Server::playlists_properties_changed`] are used
439    /// to emit `PropertiesChanged` for the `TrackList` or `Playlists`
440    /// interfaces respectively.
441    pub async fn properties_changed(
442        &self,
443        properties: impl IntoIterator<Item = Property>,
444    ) -> Result<()> {
445        let mut root_changed = HashMap::new();
446        let mut player_changed = HashMap::new();
447
448        for property in properties.into_iter() {
449            insert_property!(
450                property, Property =>
451                root_changed, CanQuit;
452                root_changed, Fullscreen;
453                root_changed, CanSetFullscreen;
454                root_changed, CanRaise;
455                root_changed, HasTrackList;
456                root_changed, Identity;
457                root_changed, DesktopEntry;
458                root_changed, SupportedUriSchemes;
459                root_changed, SupportedMimeTypes;
460                player_changed, PlaybackStatus;
461                player_changed, LoopStatus;
462                player_changed, Rate;
463                player_changed, Shuffle;
464                player_changed, Metadata;
465                player_changed, Volume;
466                player_changed, MinimumRate;
467                player_changed, MaximumRate;
468                player_changed, CanGoNext;
469                player_changed, CanGoPrevious;
470                player_changed, CanPlay;
471                player_changed, CanPause;
472                player_changed, CanSeek
473            );
474        }
475
476        if !root_changed.is_empty() {
477            self.properties_changed_inner::<RawRootInterface<T>>(root_changed, &[])
478                .await?;
479        }
480
481        if !player_changed.is_empty() {
482            self.properties_changed_inner::<RawPlayerInterface<T>>(player_changed, &[])
483                .await?;
484        }
485
486        Ok(())
487    }
488
489    async fn new_inner(
490        bus_name_suffix: &str,
491        imp: T,
492        builder_ext_func: impl FnOnce(conn::Builder<'_>, Arc<T>) -> Result<conn::Builder<'_>>
493            + Send
494            + Sync
495            + 'static,
496    ) -> Result<Self> {
497        let imp = Arc::new(imp);
498        let bus_name =
499            OwnedWellKnownName::try_from(format!("{}{}", BUS_NAME_PREFIX, bus_name_suffix))?;
500
501        let connection_builder = conn::Builder::session()?
502            .name(&bus_name)?
503            .serve_at(
504                OBJECT_PATH,
505                RawRootInterface {
506                    imp: Arc::clone(&imp),
507                },
508            )?
509            .serve_at(
510                OBJECT_PATH,
511                RawPlayerInterface {
512                    imp: Arc::clone(&imp),
513                },
514            )?;
515        let connection = builder_ext_func(connection_builder, Arc::clone(&imp))?
516            .build()
517            .await?;
518
519        Ok(Self {
520            connection,
521            imp,
522            bus_name,
523        })
524    }
525
526    /// Returns the bus name of the server.
527    pub fn bus_name(&self) -> &WellKnownName<'_> {
528        self.bus_name.inner()
529    }
530
531    /// Releases the bus name of the server.
532    ///
533    /// The bus name is released when the server is dropped. But if you want to
534    /// release it manually, you can call this method.
535    ///
536    /// Unless an error is encountered, returns `Ok(true)` if name was
537    /// previously registered with the bus and it has now been successfully
538    /// deregistered, `Ok(false)` if name was not previously registered or
539    /// already deregistered.
540    pub async fn release_bus_name(&self) -> Result<bool> {
541        self.connection.release_name(self.bus_name()).await
542    }
543
544    async fn properties_changed_inner<I>(
545        &self,
546        changed_properties: HashMap<&str, Value<'_>>,
547        invalidated_properties: &[&str],
548    ) -> Result<()>
549    where
550        I: Interface,
551    {
552        self.emit_inner::<fdo::Properties>(
553            "PropertiesChanged",
554            &(I::name(), changed_properties, invalidated_properties),
555        )
556        .await
557    }
558
559    async fn emit_inner<I>(
560        &self,
561        signal_name: &str,
562        body: &(impl Serialize + DynamicType),
563    ) -> Result<()>
564    where
565        I: Interface,
566    {
567        self.connection
568            .emit_signal(
569                None::<BusName<'_>>,
570                OBJECT_PATH,
571                I::name(),
572                signal_name,
573                body,
574            )
575            .await
576    }
577}

595    /// Emits the given signal on the `TrackList` interface.
596    pub async fn track_list_emit(&self, signal: TrackListSignal) -> Result<()> {
597        match signal {
598            TrackListSignal::TrackListReplaced {
599                tracks,
600                current_track,
601            } => {
602                self.emit_inner::<RawTrackListInterface<T>>(
603                    "TrackListReplaced",
604                    &(tracks, current_track),
605                )
606                .await?;
607            }
608            TrackListSignal::TrackAdded {
609                metadata,
610                after_track,
611            } => {
612                self.emit_inner::<RawTrackListInterface<T>>("TrackAdded", &(metadata, after_track))
613                    .await?;
614            }
615            TrackListSignal::TrackRemoved { track_id } => {
616                self.emit_inner::<RawTrackListInterface<T>>("TrackRemoved", &(track_id,))
617                    .await?;
618            }
619            TrackListSignal::TrackMetadataChanged { track_id, metadata } => {
620                self.emit_inner::<RawTrackListInterface<T>>(
621                    "TrackMetadataChanged",
622                    &(track_id, metadata),
623                )
624                .await?;
625            }
626        }
627
628        Ok(())
629    }
630
631    /// Emits the `PropertiesChanged` signal for the given properties.
632    ///
633    /// This categorizes the property in the `changed` or `invalidated`
634    /// properties as defined by the spec.
635    pub async fn track_list_properties_changed(
636        &self,
637        properties: impl IntoIterator<Item = TrackListProperty>,
638    ) -> Result<()> {
639        let mut changed = HashMap::new();
640        let mut invalidated = Vec::new();
641
642        for property in properties.into_iter() {
643            match property {
644                TrackListProperty::Tracks => {
645                    invalidated.push("Tracks");
646                }
647                TrackListProperty::CanEditTracks(can_edit_tracks) => {
648                    changed.insert("CanEditTracks", Value::new(can_edit_tracks));
649                }
650            }
651        }
652
653        if !changed.is_empty() || !invalidated.is_empty() {
654            self.properties_changed_inner::<RawTrackListInterface<T>>(changed, &invalidated)
655                .await?;
656        }
657
658        Ok(())
659    }
660}

731    /// implementation, which must implement [`TrackListInterface`] and
732    /// [`PlaylistsInterface`] in addition to [`RootInterface`] and
733    /// [`PlayerInterface`].
734    ///
735    /// See also [`Server::new`].
736    pub async fn new_with_all(bus_name_suffix: &str, imp: T) -> Result<Self> {
737        Self::new_inner(bus_name_suffix, imp, |builder, imp| {
738            builder
739                .serve_at(
740                    OBJECT_PATH,
741                    RawTrackListInterface {
742                        imp: Arc::clone(&imp),
743                    },
744                )?
745                .serve_at(OBJECT_PATH, RawPlaylistsInterface { imp })
746        })
747        .await
748    }
749}
